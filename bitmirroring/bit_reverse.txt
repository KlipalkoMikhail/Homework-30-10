char * bit_reverse(char *str)               // зеркальное отображение битов, параметр - массив бит
{
    char tmp;                               // временная переменная
    int size = 8*sizeof(int);               // размер в 32 бита
    for(int i = 0; i < (size)/2; i++)       // повышаем индекс на 1 и меняем биты в симметричных ячейках местами, дойдя до середины прекратим обмен
    {
        tmp = str[i];                       // сохраняем один бит
        str[i] = str[size - i - 1];         // меняем местами
        str[size - i - 1] = tmp;            // используем наши сохраненки
    }
    str[size] = '\0';                       // символ конца строки
    return str;                             // возвращаем массив бит
}
